public static void calcDet(int[][] matriz){
        int dp = 0;
        int ds = 0;
        int n = matriz[0].length;
        int colP = 0;
        int colS = n-1;
        for(int i = 0; i<n; i++){
            int diagonalP = 1;
            int diagonalS = 1;
             for (int linha = 0; linha < n; linha++) {
                 if(colP == n){
                     colP = 0;
                 }
                 else if(colP < 0){
                     colP = n-1;
                 }
                 if(colS == n){
                     colS = 0;
                 }
                 else if(colS < 0){
                     colS = n-1;
                 }
                 diagonalS *= matriz[linha][colS];
                 diagonalP *= matriz[linha][colP];
                 colP ++;
                 colS --;
             }
             ds += diagonalS;
             dp += diagonalP;
             colP --;
             colS ++;
        }
        int determinante = dp-ds;
        if(determinante < 0){
            determinante *=-1;
        }
       
        System.out.println("Diagonal principal: "+dp);
        System.out.println("Diagonal secundÃ¡ria: "+ds);
        System.out.println("determinante: "+determinante);
}
