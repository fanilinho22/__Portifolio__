    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        float capital = Float.parseFloat(CAPITAL.getText());
        int tempo = Integer.parseInt(TEMPO.getText());
        float taxa = Float.parseFloat(TAXA.getText());
        float juros = capital * taxa * tempo;
        float montante = capital + (juros);

        String mensagem = "O valor do montante é R$" + montante + "\nO valor dos juros é R$" + juros;

        JOptionPane.showMessageDialog(null, mensagem);
    }

    private void TAXAKeyReleased(java.awt.event.KeyEvent evt) {                                 
        if (!TAXA.getText().equals(null) || !TAXA.getText().equals("")) {
            double nieto = Double.parseDouble(TAXA.getText());
            decimal.setText((nieto/100)+"");
            //decimal.setText(TAXA.getText().substring(0, 1) + "," + TAXA.getText().substring(1));
        }
        else{
            decimal.setText("");
        }
    }
